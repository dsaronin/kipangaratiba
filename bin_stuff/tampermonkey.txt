// ==UserScript==
// @name         Jitsi Auto Join/Hangup (v1.8)
// @namespace    http://tampermonkey.net/
// @version      1.8.1
// @description  Auto joins; auto-hangsup; based on room_name-to-duration.
// @author       Angalia Developer
// @match        https://meet.jit.si/*
// @grant        none
// @run-at       document-idle
// ==/UserScript==

(function() {
    'use strict';
    
    // --- APP VERSION ---
    // When updating the script over SSH, increment this version string directly
    const SCRIPT_VERSION = '1.8.1';

    console.log(`[Angalia Script] v${SCRIPT_VERSION} START`);   

    // --- CONFIGURATION ---
    const DEFAULT_DURATION_SECONDS = 3600; // 60 minutes

    // Hash map of Room Name (MUST BE LOWERCASE) => Duration in Seconds
    const DURATION_MAP = {
        "schfrry102visitsa": 4500,  // 75 * 60
        "ellesgantesgs1976": 7200,  // 120 * 60
        "shortselmasession5145": 1800   // 30 * 60
    };

    // --- XPATH SELECTORS ---
    const xpaths = {
        join: "//div[@role='button' and contains(translate(., 'ABCDEFGHIJKLMNOPQRSTUVWXYZ', 'abcdefghijklmnopqrstuvwxyz'), 'join meeting')]",
        // login: "//button[@data-testid='lobby.loginButton']",
        login: "//button[@data-testid='lobby.loginButton' or contains(., 'Log-in')]",
        hangup: "//div[contains(@class, 'toolbox-icon') and contains(@class, 'hangup-button')]",
        incallMic: "//div[@role='button' and (@aria-label='Mute microphone' or @aria-label='Unmute microphone')]",
        incallCam: "//div[@role='button' and (@aria-label='Stop camera' or @aria-label='Start camera')]"
    };

    // --- STATE MACHINE FOR POLLING ---
    let pollInterval = null;
    let safetyTimeout = null;
    let pollerState = null;
    let mediaChecked = false;

    // --- TELEMETRY HELPERS ---
    function getElementTelemetry(node, selectorName) {
        if (!node) return `${selectorName}: NOT_IN_DOM`;
        
        try {
            const style = window.getComputedStyle(node);
            const isVisible = !!(node.offsetWidth || node.offsetHeight || node.getClientRects().length) && 
                              style.display !== 'none' && 
                              style.visibility !== 'hidden' && 
                              style.opacity !== '0';
            
            // Extract identifier strings to trace exactly what changed
            const attrs = {
                visible: isVisible ? 'VISIBLE' : 'HIDDEN'
            };
            return `${selectorName}:${attrs.visible}`;
        } catch (e) {
            return `${selectorName}: METRIC_FAILED (${e.message})`;
        }
    }

  function startPoller(initialState) {
        if (pollInterval) clearInterval(pollInterval);
        if (safetyTimeout) clearTimeout(safetyTimeout);

        pollerState = initialState;
        let targetXPath = '';
        console.log(`[Angalia Script] Starting poller for: ${pollerState}`);

        // --- Configure the target for simple states ---
        if (pollerState === 'POLL_FOR_JOIN') {
            targetXPath = xpaths.join;
        } else if (pollerState === 'POLL_FOR_HANGUP') {
            targetXPath = xpaths.hangup;
        } else if (pollerState !== 'POLL_FOR_POST_JOIN' && pollerState !== 'POLL_FOR_MEDIA_RESTORE') {
            console.log(`[Angalia Script] Unknown state, stopping.`);
            return;
        }

        // --- Safety Timeout ---
        safetyTimeout = setTimeout(() => {
            console.log(`[Angalia Script] Poller for ${pollerState} TIMED OUT after 30s.`);
            clearInterval(pollInterval);
        }, 600000);

        // --- Polling Interval ---
        pollInterval = setInterval(() => {
            try {
                
                // INSERT:
                // --- SPECIAL LOGIC FOR MEDIA RESTORE ---
                if (pollerState === 'POLL_FOR_MEDIA_RESTORE') {
                    let micNode = document.evaluate(xpaths.incallMic, document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
                    let camNode = document.evaluate(xpaths.incallCam, document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;

                    if (micNode && camNode) {
                        console.log('[Angalia Script] Toolbar mounted. Evaluating media stream states...');
                        
                        let micLabel = micNode.getAttribute('aria-label') || '';
                        let camLabel = camNode.getAttribute('aria-label') || '';

                        if (micLabel.toLowerCase().includes('unmute')) {
                            console.log('[Angalia Script] Microphone is muted. Firing unmute trigger.');
                            micNode.click();
                        } else {
                            console.log('[Angalia Script] Microphone is already active.');
                        }

                        if (camLabel.toLowerCase().includes('start')) {
                            console.log('[Angalia Script] Camera is muted. Firing stream start trigger.');
                            camNode.click();
                        } else {
                            console.log('[Angalia Script] Camera is already active.');
                        }

                        clearInterval(pollInterval);
                        clearTimeout(safetyTimeout);
                        mediaChecked = true;
                        startPoller('POLL_FOR_POST_JOIN');
                    }
                }

                // --- SPECIAL LOGIC FOR POST-JOIN ---
                // This state waits for EITHER the lobby OR the meeting to load.
                if (pollerState === 'POLL_FOR_POST_JOIN') {
                    let loginBtn = document.evaluate(xpaths.login, document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
                    let hangupBtn = document.evaluate(xpaths.hangup, document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
                    
                    // Look for the container wrapper that indicates the staging area/lobby is obscuring the interface
                    let isScreenBlocked = !!document.querySelector('.premeeting-screen');

                    // Continuous Trace: Log structural attributes on every 500ms pass
                    console.log(`[Angalia Trace] ${getElementTelemetry(loginBtn, 'Login')} | ${getElementTelemetry(hangupBtn, 'Hangup')} | LobbyScreenOverlay:${isScreenBlocked ? 'ACTIVE' : 'ABSENT'}`);

                    if (loginBtn) {
                        console.log(`[Angalia Script] Security Gate Active: Found POLL_FOR_LOGIN (lobby). Extracting React handler...`);
                        
                        try {
                            const reactPropsKey = Object.keys(loginBtn).find(key => 
                                key.startsWith('__reactProps') || key.startsWith('__reactEventHandlers')
                            );

                           // Bypassing browser popups: Intercept window.open and redirect the current window context instead
                            const originalWindowOpen = window.open;
                            window.open = function(url) {
                                console.log('[Angalia Script] Caught popup allocation attempt. Redirecting current frame context directly to authentication target URL:', url);
                                window.location.href = url;
                                return window; // Return current window reference to prevent Jitsi from throwing an undefined error
                            };

                            if (reactPropsKey && loginBtn[reactPropsKey] && typeof loginBtn[reactPropsKey].onClick === 'function') {
                                console.log('[Angalia Script] Executing native React onClick handler...');
                                loginBtn[reactPropsKey].onClick({ preventDefault: () => {}, stopPropagation: () => {} });
                            } else {
                                console.log('[Angalia Script] React properties missing, executing DOM fallback click.');
                                loginBtn.click();
                            }

                            // Restore original native layout method immediately after invocation pass
                            setTimeout(() => { window.open = originalWindowOpen; }, 100); 

                        } catch (reactError) {
                            console.error('[Angalia Script] Exception during React execution:', reactError);
                            loginBtn.click(); // Fallback execution
                        }

                        clearInterval(pollInterval);
                        clearTimeout(safetyTimeout);
                        console.log('[Angalia Script] Join/Login sequence complete.');
                    } else if (hangupBtn && !isScreenBlocked) {
                        if (!mediaChecked) {
                            console.log(`[Angalia Script] Successfully joined meeting. Advancing to media restore phase.`);
                            clearInterval(pollInterval);
                            clearTimeout(safetyTimeout);
                            startPoller('POLL_FOR_MEDIA_RESTORE');
                        } else {
                            console.log(`[Angalia Script] Call stable and media restore complete. Active tracking loop idling.`);
                            clearInterval(pollInterval);
                            clearTimeout(safetyTimeout);
                        }
                    }
                    
                // --- STANDARD LOGIC for other states ---
                } else {

                    // Direct-entry bypass if call loads directly without a landing page button
                    if (pollerState === 'POLL_FOR_JOIN') {
                        let joinBtnPresent = document.evaluate(xpaths.join, document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
                        let activeCallActive = document.evaluate(xpaths.incallMic, document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
                        
                        // Only bypass if in-call elements exist AND the join button is completely missing
                        if (activeCallActive && !joinBtnPresent) {
                            console.log('[Angalia Script] Direct entry verified (Join button absent). Transitioning to POLL_FOR_POST_JOIN.');
                            clearInterval(pollInterval);
                            clearTimeout(safetyTimeout);
                            startPoller('POLL_FOR_POST_JOIN');
                            return;
                        }
                    }                    

                    let result = document.evaluate(targetXPath, document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null);
                    let button = result.singleNodeValue;

                    if (button) {
                        console.log(`[Angalia Script] Found and clicking: ${pollerState}`);
                        button.click();
                        clearInterval(pollInterval);
                        clearTimeout(safetyTimeout);

                        if (pollerState === 'POLL_FOR_JOIN') {
                            console.log('[Angalia Script] Join clicked. Now polling for Post-Join (Lobby or Hangup).');
                            startPoller('POLL_FOR_POST_JOIN');
                        } else if (pollerState === 'POLL_FOR_HANGUP') {
                            console.log('[Angalia Script] Hangup complete.');
                        }
                    }
                }
            } catch (e) {
                console.error('[Angalia Script] Poller error:', e);
                clearInterval(pollInterval);
              clearTimeout(safetyTimeout);
            }
        }, 500);
    }

    // --- HANGUP TIMER ---
    function setupHangupTimer() {
        let hangupDuration = DEFAULT_DURATION_SECONDS;
        try {
            // Get room name from URL path and force to lowercase to maintain case-insensitivity
            const rawRoomName = window.location.pathname.substring(1);
            const roomNameLookup = rawRoomName.toLowerCase();

            if (DURATION_MAP.hasOwnProperty(roomNameLookup)) {
                hangupDuration = DURATION_MAP[roomNameLookup];
                console.log(`[Angalia Script] Found room '${rawRoomName}' (matched as '${roomNameLookup}'). Setting duration to ${hangupDuration}s.`);
            } else {
                console.log(`[Angalia Script] Room '${rawRoomName}' not in configuration map. Using default ${hangupDuration}s.`);
            }
            const durationMs = hangupDuration * 1000;

            setTimeout(() => {
                console.log(`[Angalia Script] HANGUP TIMER FIRED after ${hangupDuration}s. Starting hangup poller.`);
                startPoller('POLL_FOR_HANGUP');
            }, durationMs);

        } catch (e) {
            console.error('[Angalia Script] Failed to parse room/duration:', e);
        }
    }

    // --- START ---
    startPoller('POLL_FOR_JOIN');
    setupHangupTimer();

})();
